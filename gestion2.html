⚙️ Configurar EcoReportes
Para que los botones Iniciar y Finalizar escriban en tu Google Sheet, necesitas un Apps Script desplegado como web app.
1. Abre tu Google Sheet → Extensiones → Apps Script
2. Borra el contenido existente y pega el código de abajo
3. Guarda (Ctrl+S) y haz clic en Implementar → Nueva implementación
4. Tipo: Aplicación web | Ejecutar como: Yo | Acceso: Cualquier persona
5. Autoriza los permisos y copia la URL de implementación
6. Ve a la pestaña 2. URL & Columnas y pega la URL
// ══════════════════════════════════════════════ // EcoReportes — Google Apps Script // Pega este código en Extensiones → Apps Script // ══════════════════════════════════════════════ var SHEET_NAME = "Hoja1"; // Cambia si tu hoja tiene otro nombre function doGet(e) { return manejarPeticion(e); } function doPost(e) { return manejarPeticion(e); } function manejarPeticion(e) { try { var params = e.parameter; if (params.action === "updateEstado") { return respuesta(actualizarEstado(params)); } return respuesta({ success: false, error: "Accion desconocida" }); } catch (err) { return respuesta({ success: false, error: err.message }); } } function respuesta(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function actualizarEstado(params) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "Hoja no encontrada: " + SHEET_NAME }; var rowId = params.rowId; var colId = params.colId; var colEstado = params.colEstado; var nuevoEstado = params.estado; var data = sheet.getDataRange().getValues(); var headers = data[0].map(function(h){ return h.toString().toLowerCase().trim(); }); // Buscar o crear columna de estado var idxEstado = headers.indexOf(colEstado.toLowerCase().trim()); if (idxEstado === -1) { idxEstado = headers.length; sheet.getRange(1, idxEstado + 1).setValue(colEstado); } var targetRow = -1; // Buscar por columna ID personalizada if (colId && colId !== "id") { var idxId = headers.indexOf(colId.toLowerCase().trim()); if (idxId !== -1) { for (var i = 1; i < data.length; i++) { if (data[i][idxId].toString() === rowId.toString()) { targetRow = i + 1; break; } } } } // Fallback: número de fila directo if (targetRow === -1) { var rowNum = parseInt(rowId); if (!isNaN(rowNum) && rowNum >= 2 && rowNum <= data.length + 1) { targetRow = rowNum; } } if (targetRow === -1) return { success: false, error: "Fila no encontrada: " + rowId }; sheet.getRange(targetRow, idxEstado + 1).setValue(nuevoEstado); return { success: true, row: targetRow, estado: nuevoEstado }; }
