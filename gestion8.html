⚙️ Configurar EcoReportes
Para que los botones Iniciar y Finalizar escriban en tu Google Sheet, necesitas un Apps Script desplegado como web app.
1. Abre tu Google Sheet → Extensiones → Apps Script
2. Borra el contenido y pega el código de abajo
3. Guarda (Ctrl+S) → Implementar → Nueva implementación
4. Tipo: Aplicación web | Ejecutar como: Yo | Acceso: Cualquier persona
5. Autoriza permisos y copia la URL de implementación
6. Ve a pestaña 2. URL & Columnas y pega la URL
// ══════════════════════════════════════════════ // EcoReportes — Google Apps Script // ══════════════════════════════════════════════ var SHEET_NAME = "Hoja 1"; // nombre exacto de tu hoja function doGet(e) { return manejarPeticion(e); } function doPost(e) { return manejarPeticion(e); } function manejarPeticion(e) { try { var p = e.parameter; if (p.action === "updateEstado") { return respuesta(actualizarEstado(p)); } return respuesta({ success: false, error: "Accion desconocida" }); } catch (err) { return respuesta({ success: false, error: err.message }); } } function respuesta(obj) { var out = ContentService.createTextOutput(JSON.stringify(obj)); out.setMimeType(ContentService.MimeType.JSON); return out; } function actualizarEstado(p) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "Hoja no encontrada: " + SHEET_NAME }; var data = sheet.getDataRange().getValues(); var headers = data[0].map(function(h){ return h.toString().toLowerCase().trim(); }); // Buscar o crear columna de estado var idxEstado = headers.indexOf(p.colEstado.toLowerCase().trim()); if (idxEstado === -1) { idxEstado = headers.length; sheet.getRange(1, idxEstado + 1).setValue(p.colEstado); data[0][idxEstado] = p.colEstado.toLowerCase().trim(); } var targetRow = -1; // Buscar por columna ID si existe if (p.colId && p.colId !== "id") { var idxId = headers.indexOf(p.colId.toLowerCase().trim()); if (idxId !== -1) { for (var i = 1; i < data.length; i++) { if (data[i][idxId].toString() === p.rowId.toString()) { targetRow = i + 1; break; } } } } // Fallback: número de fila directo if (targetRow === -1) { var rowNum = parseInt(p.rowId); if (!isNaN(rowNum) && rowNum >= 2 && rowNum <= data.length + 1) { targetRow = rowNum; } } if (targetRow === -1) return { success: false, error: "Fila no encontrada: " + p.rowId }; sheet.getRange(targetRow, idxEstado + 1).setValue(p.estado); SpreadsheetApp.flush(); return { success: true, row: targetRow, estado: p.estado }; }
