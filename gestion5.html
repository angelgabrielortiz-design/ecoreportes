⚙️ Configurar EcoReportes
Para que los botones funcionen necesitas el Apps Script desplegado.
1. Abre tu Google Sheet → Extensiones → Apps Script
2. Borra todo y pega el código de abajo
3. Guarda y Implementar → Nueva implementación
4. Tipo: Aplicación web | Ejecutar como: Yo | Acceso: Cualquier persona
// ══════════════════════════════════════════════ // EcoReportes — Apps Script (VERSIÓN MEJORADA) // ══════════════════════════════════════════════ var SHEET_NAME = "Hoja1"; function doGet(e) { return manejarPeticion(e); } function doPost(e) { return manejarPeticion(e); } function manejarPeticion(e) { try { var params = e.parameter; if (params.action === "updateEstado") { return respuesta(actualizarEstado(params)); } return respuesta({ success: false, error: "Acción desconocida" }); } catch (err) { return respuesta({ success: false, error: err.message }); } } function respuesta(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function actualizarEstado(params) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: false, error: "Hoja no encontrada" }; var rowId = params.rowId; var colId = params.colId || "id"; var colEstado = params.colEstado || "estado"; var nuevoEstado = params.estado; var data = sheet.getDataRange().getValues(); var headers = data[0].map(h => h.toString().toLowerCase().trim()); // Buscar o crear columna de estado var idxEstado = headers.indexOf(colEstado.toLowerCase().trim()); if (idxEstado === -1) { idxEstado = headers.length; sheet.getRange(1, idxEstado + 1).setValue(colEstado); } var targetRow = -1; // Buscar por columna ID if (colId && colId !== "id") { var idxId = headers.indexOf(colId.toLowerCase().trim()); if (idxId !== -1) { for (var i = 1; i < data.length; i++) { if (String(data[i][idxId]).trim() === String(rowId).trim()) { targetRow = i + 1; break; } } } } // Fallback: usar número de fila if (targetRow === -1) { var rowNum = parseInt(rowId); if (!isNaN(rowNum) && rowNum >= 2) targetRow = rowNum; } if (targetRow === -1) return { success: false, error: "Fila no encontrada: " + rowId }; sheet.getRange(targetRow, idxEstado + 1).setValue(nuevoEstado); return { success: true, row: targetRow, estado: nuevoEstado, timestamp: new Date().toISOString() }; }
